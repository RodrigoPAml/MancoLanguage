function find_greater_number(integer a<>, integer n, integer& greater)
    integer i = 0
    while i < n
        if greater < a[i]
            greater = a[i]
        end
        i = i + 1
    end
end

function find_sum(integer a<>, integer n, integer& sum)
    integer i = 0
    while i < n
        sum = sum + a[i]
        i = i + 1
    end
end

function insert(integer a<>, integer n)
    integer i = 0
    while i < n
       a[i] = i 
       i = i + 1

end
